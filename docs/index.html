CellEd

Simple cell editor

Dynamic size grid with readonly column

var grid1 = new CellEd.Grid('#grid1', {
    cols: ['A', 'B'],
    rows: [
        [{ value: 1, readonly: true}, 2],
        [{ value: 1, readonly: true}, 2],
        [{ value: 1, readonly: true}, 2],
    ],
    canAddRows: true,
});
grid1.addRows([[{ value: 'added', readonly: true}, 'added']]);

Fixed size grid

var grid2 = new CellEd.Grid('#grid2');
grid2.init({
    cols: ['A', 'B', 'C'],
    rows: [
        [1, 2, 3],
        [1, 2, 3],
        [1, 2, 3],
    ],
});
